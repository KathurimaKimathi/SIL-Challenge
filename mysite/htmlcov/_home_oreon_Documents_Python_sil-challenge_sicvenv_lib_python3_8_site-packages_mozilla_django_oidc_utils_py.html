61def add_state_and_nonce_to_session(request, state, params): 

62 """ 

63 Stores the `state` and `nonce` parameters in a session dictionary including the time when it 

64 was added. The dictionary can contain multiple state/nonce combinations to allow parallel 

65 logins with multiple browser sessions. 

66 To keep the session space to a reasonable size, the dictionary is kept at 50 state/nonce 

67 combinations maximum. 

68 """ 

69 nonce = params.get('nonce') 

70 

71 # Store Nonce with the State parameter in the session "oidc_states" dictionary. 

72 # The dictionary can store multiple State/Nonce combinations to allow parallel 

73 # authentication flows which would otherwise overwrite State/Nonce values! 

74 # The "oidc_states" dictionary uses the state as key and as value a dictionary with "nonce" 

75 # and "added_on". "added_on" contains the time when the state was added to the session. 

76 # With this value, the oldest element can be found and deleted from the session. 

77 if 'oidc_states' not in request.session or \ 

78 not isinstance(request.session['oidc_states'], dict): 

79 request.session['oidc_states'] = {} 

80 

81 # Make sure that the State/Nonce dictionary in the session does not get too big. 

82 # If the number of State/Nonce combinations reaches a certain threshold, remove the oldest 

83 # state by finding out 

84 # which element has the oldest "add_on" time. 

85 limit = import_from_settings('OIDC_MAX_STATES', 50) 

86 if len(request.session['oidc_states']) >= limit: 

87 LOGGER.info( 

88 'User has more than {} "oidc_states" in his session, ' 

89 'deleting the oldest one!'.format(limit) 

90 ) 

91 oldest_state = None 

92 oldest_added_on = time.time() 

93 for item_state, item in request.session['oidc_states'].items(): 

94 if item['added_on'] < oldest_added_on: 

95 oldest_state = item_state 

96 oldest_added_on = item['added_on'] 

97 if oldest_state: 

98 del request.session['oidc_states'][oldest_state] 

99 

100 request.session['oidc_states'][state] = { 

101 'nonce': nonce, 

102 'added_on': time.time(), 

103 }
